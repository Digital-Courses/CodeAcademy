# --- Framework Principal ---
Django==4.2.*                        # El framework web (como el motor del carro)

# --- Base de Datos ---
psycopg2-binary==2.9.*               # Driver para conectar Django con PostgreSQL

djangorestframework                  # DRF: convierte Django en una API REST

django-cors-headers                  # Permite que React (otro dominio) hable con Django

# --- Utilidades ---
python-dotenv==1.0.*                 # Lee variables del archivo .env
Pillow                               # Procesamiento de imágenes (para las portadas de productos)
